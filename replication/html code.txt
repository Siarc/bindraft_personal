<script>
    document.addEventListener("DOMContentLoaded", () => {
        // Desktop dropdown navigation functionality
        const navItems = document.querySelectorAll(".nav-middle li[data-dropdown]");
        const dropdowns = document.querySelectorAll(".dropdown-list-wrapper");

        navItems.forEach((item) => {
            const dropdownId = item.getAttribute("data-dropdown");
            const dropdown = document.getElementById(dropdownId);

            item.addEventListener("click", (e) => {
                e.stopPropagation();
                const isOpen = dropdown.classList.contains("show");

                // Close all dropdowns first
                dropdowns.forEach((d) => d.classList.remove("show"));
                navItems.forEach((li) => li.classList.remove("active"));

                // Toggle current one
                if (!isOpen) {
                    dropdown.classList.add("show");
                    item.classList.add("active");
                }
            });
        });

        // Mobile dropdown/accordion functionality
        const dropdownWrappers = document.querySelectorAll(".moblie-dropdown-wrapper");

        dropdownWrappers.forEach(wrapper => {
            wrapper.addEventListener("click", () => {
                const parentLi = wrapper.closest("li");
                const dropdown = parentLi.querySelector(".dropdown-mobile-list-wrapper");
                const arrow = wrapper.querySelector(".drop-icon svg");

                // Close any open dropdowns except the one clicked
                document.querySelectorAll(".dropdown-mobile-list-wrapper.show").forEach(
                    openDropdown => {
                        if (openDropdown !== dropdown) {
                            openDropdown.classList.remove("show");
                            openDropdown.style.maxHeight = null;
                            openDropdown.previousElementSibling
                                .querySelector(".drop-icon svg")
                                .style.transform = "rotate(0deg)";
                        }
                    });

                // Toggle current dropdown
                if (dropdown.classList.contains("show")) {
                    dropdown.classList.remove("show");
                    dropdown.style.maxHeight = null;
                    arrow.style.transform = "rotate(0deg)";
                } else {
                    dropdown.classList.add("show");
                    let height = dropdown.scrollHeight + 20;
                    dropdown.style.maxHeight = height + "px";
                    arrow.style.transform = "rotate(180deg)";
                }
            });
        });

        // Function to close all mobile accordions
        function closeMobileAccordions() {
            document.querySelectorAll(".dropdown-mobile-list-wrapper.show").forEach(dropdown => {
                dropdown.classList.remove("show");
                dropdown.style.maxHeight = null;
                const arrow = dropdown.previousElementSibling?.querySelector(".drop-icon svg");
                if (arrow) {
                    arrow.style.transform = "rotate(0deg)";
                }
            });
        }

        // Function to close all desktop dropdowns
        function closeDesktopDropdowns() {
            dropdowns.forEach((d) => d.classList.remove("show"));
            navItems.forEach((li) => li.classList.remove("active"));
        }

        // Mobile menu functionality using your actual selectors
        const burgerButton = document.querySelector(".menu-btn-1");
        const mobileMenu = document.querySelector(".nav-menu");

        // Function to monitor mobile menu state and close accordions when menu closes
        function setupMobileMenuObserver() {
            if (!mobileMenu) return;

            const observer = new MutationObserver((mutations) => {
                mutations.forEach((mutation) => {
                    if (mutation.type === 'attributes' && mutation.attributeName === 'class') {
                        const isMenuOpen = mobileMenu.classList.contains("show") || mobileMenu
                            .classList.contains("active");
                        if (!isMenuOpen) {
                            // Menu is closed, ensure accordions are closed
                            closeMobileAccordions();
                        }
                    }
                });
            });

            observer.observe(mobileMenu, {
                attributes: true,
                attributeFilter: ['class']
            });

            // Also check initial state
            const isMenuOpen = mobileMenu.classList.contains("show") || mobileMenu.classList.contains("active");
            if (!isMenuOpen) {
                closeMobileAccordions();
            }
        }

        if (burgerButton && mobileMenu) {
            // Setup observer to watch menu state changes
            setupMobileMenuObserver();

            burgerButton.addEventListener("click", (e) => {
                e.stopPropagation();
                const isMenuOpen = mobileMenu.classList.contains("show") || mobileMenu.classList.contains(
                    "active");

                if (isMenuOpen) {
                    // If menu is open, close accordions first, then menu
                    closeMobileAccordions();
                    setTimeout(() => {
                        mobileMenu.classList.remove("show", "active");
                        burgerButton.classList.remove("active");
                        burgerButton.setAttribute("aria-expanded", "false");
                    }, 50); // Small delay to see accordion close first
                } else {
                    // If menu is closed, ensure accordions are closed and open menu
                    closeMobileAccordions();
                    mobileMenu.classList.add("show");
                    burgerButton.classList.add("active");
                    burgerButton.setAttribute("aria-expanded", "true");
                }
            });
        }

        // Close everything when clicking outside
        document.addEventListener("click", (e) => {
            // Don't close if clicking inside mobile menu or burger button
            if (mobileMenu && (mobileMenu.contains(e.target) || (burgerButton && burgerButton.contains(e
                    .target)))) {
                return;
            }

            // Close desktop dropdowns
            closeDesktopDropdowns();

            // Close mobile menu and accordions if clicking outside
            if (mobileMenu && !mobileMenu.contains(e.target)) {
                const wasMenuOpen = mobileMenu.classList.contains("show") || mobileMenu.classList.contains(
                    "active");
                if (wasMenuOpen) {
                    closeMobileAccordions();
                    mobileMenu.classList.remove("show", "active");
                    if (burgerButton) {
                        burgerButton.classList.remove("active");
                        burgerButton.setAttribute("aria-expanded", "false");
                    }
                }
            }
        });

        // Optional: Close mobile menu on window resize (if switching to desktop)
        window.addEventListener("resize", () => {
            if (window.innerWidth > 768) { // Adjust breakpoint as needed
                const wasMenuOpen = mobileMenu && (mobileMenu.classList.contains("show") || mobileMenu
                    .classList.contains("active"));
                if (wasMenuOpen) {
                    closeMobileAccordions();
                    mobileMenu.classList.remove("show", "active");
                    if (burgerButton) {
                        burgerButton.classList.remove("active");
                        burgerButton.setAttribute("aria-expanded", "false");
                    }
                }
            }
        });

        // Optional: Handle escape key to close everything
        document.addEventListener("keydown", (e) => {
            if (e.key === "Escape") {
                closeDesktopDropdowns();
                const wasMenuOpen = mobileMenu && (mobileMenu.classList.contains("show") || mobileMenu
                    .classList.contains("active"));
                if (wasMenuOpen) {
                    closeMobileAccordions();
                    mobileMenu.classList.remove("show", "active");
                    if (burgerButton) {
                        burgerButton.classList.remove("active");
                        burgerButton.setAttribute("aria-expanded", "false");
                    }
                }
            }
        });
    });

    <script>
    $(document).ready(function() {
        const totalSlides = 4;
        let playingVideoIndex = null;

        const $desktopCards = $('[data-card-index]');
        const $desktopVideos = $desktopCards.find('.card-video');
        const $desktopButtons = $desktopCards.find('.play-button');

        const $carousel = $('#carousel-track');
        const $carouselVideos = $carousel.find('.card-video');
        const $carouselButtons = $carousel.find('.mobile-play-button');

        // ---------------------------
        // Init Slick for Mobile/Tablet
        // ---------------------------
        function initSlick() {
            if ($(window).width() < 1024) {
                if (!$carousel.hasClass('slick-initialized')) {
                    $carousel.slick({
                        dots: true,
                        arrows: true,
                        infinite: false,
                        speed: 400,
                        slidesToShow: $(window).width() >= 768 ? 1.5 : 1,
                        slidesToScroll: $(window).width() >= 768 ? 1 : 1,
                    });

                    $carousel.on('beforeChange', function() {
                        pauseAllVideos();
                    });
                }
            } else {
                if ($carousel.hasClass('slick-initialized')) {
                    $carousel.slick('unslick');
                }
            }
        }

        initSlick();
        $(window).on('resize', function() {
            initSlick();
            pauseAllVideos();
        });

        // ---------------------------
        // Desktop hover expand/collapse
        // ---------------------------
        $desktopCards.on('mouseenter', function() {
            const index = $(this).data('card-index');
            $desktopCards.removeClass('expanded').addClass('collapsed').attr('aria-expanded', 'false');
            $(this).removeClass('collapsed').addClass('expanded').attr('aria-expanded', 'true');

            if (playingVideoIndex !== null && playingVideoIndex !== index) {
                pauseVideo(playingVideoIndex);
            }
        });

        // ---------------------------
        // Play/Pause button click
        // ---------------------------
        function togglePlayPause(index, isDesktop) {
            const video = isDesktop ? $desktopVideos.get(index) : $carouselVideos.get(index);
            const button = isDesktop ? $desktopButtons.get(index) : $carouselButtons.get(index);

            if (!video || !button) return;

            if (playingVideoIndex === index && !video.paused) {
                video.pause();
                playingVideoIndex = null;
                updateButtonState($(button), false);
            } else {
                pauseAllVideos();
                video.muted = false;
                video.play().then(() => {
                    playingVideoIndex = index;
                    updateButtonState($(button), true);
                }).catch(err => {
                    console.error('Video playback error:', err);
                });
            }
        }

        $desktopButtons.each(function(index) {
            $(this).on('click', () => togglePlayPause(index, true));
        });
        $carouselButtons.each(function(index) {
            $(this).on('click', () => togglePlayPause(index, false));
        });

        // ---------------------------
        // Helpers
        // ---------------------------
        function pauseVideo(index) {
            const dVideo = $desktopVideos.get(index);
            const cVideo = $carouselVideos.get(index);
            const dBtn = $desktopButtons.get(index);
            const cBtn = $carouselButtons.get(index);

            if (dVideo && !dVideo.paused) dVideo.pause();
            if (cVideo && !cVideo.paused) cVideo.pause();

            if (dBtn) updateButtonState($(dBtn), false);
            if (cBtn) updateButtonState($(cBtn), false);
        }

        function pauseAllVideos() {
            for (let i = 0; i < totalSlides; i++) {
                pauseVideo(i);
            }
            playingVideoIndex = null;
        }

        function updateButtonState($button, isPlaying) {
            $button.attr('aria-pressed', isPlaying.toString());
            $button.find('.play-icon').toggleClass('hidden', isPlaying);
            $button.find('.pause-icon').toggleClass('hidden', !isPlaying);
        }

        // Reset button states on video end
        $desktopVideos.each(function(index) {
            $(this).on('ended', () => {
                updateButtonState($($desktopButtons.get(index)), false);
                playingVideoIndex = null;
            });
        });
        $carouselVideos.each(function(index) {
            $(this).on('ended', () => {
                updateButtonState($($carouselButtons.get(index)), false);
                playingVideoIndex = null;
            });
        });
    });
    </script>

    function initGMSlider() {
        if ($(window).width() <= 991) {

                        breakpoint: 768,
                        settings: {
                            slidesToShow: 1,

                    }]
                });
            }
        } else {
            if ($('.gm-slider').hasClass('slick-initialized')) {
                $('.gm-slider').slick('unslick'); // go back to normal 4-column grid
            }
        }
    }
    // init on load + resize
    $(document).ready(initGMSlider);
    $(window).on('resize', initGMSlider);
    </script>
